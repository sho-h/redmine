def extract_sections(index)
  @pre_list = []
  text = self.dup
  rip_offtags text, false, false
  before = ''
  s = ''
  after = ''
  i = 0
  l = 1
  started = false
  ended = false
  text.scan(/(((?:.*?)(\A|\r?\n\s*\r?\n))(h(\d+)(#{A}#{C})\.(?::(\S+))?[ \t](.*?)$)|.*)/m).each do |all, content, lf, heading, level|
    if heading.nil?
      if ended
        after << all
      elsif started
        s << all
      else
        before << all
      end
      break
    end
    i += 1
    if ended
      after << all
    elsif i == index
      l = level.to_i
      before << content
      s << heading
      started = true
    elsif i > index
      s << content
      if level.to_i > l
        s << heading
      else
        after << heading
        ended = true
      end
    else
      before << all
    end
  end
  sections = [before.strip, s.strip, after.strip]
  sections.each {|section| smooth_offtags_without_code_highlighting section}
  sections
end

def get_section(index)
  section = extract_sections(index)[1]
  hash = Digest::MD5.hexdigest(section)
  return section, hash
end

def update_section(index, update, hash=nil)
  t = extract_sections(index)
  if hash.present? && hash != Digest::MD5.hexdigest(t[1])
    raise Redmine::WikiFormatting::StaleSectionError
  end
  t[1] = update unless t[1].blank?
  t.reject(&:blank?).join "\n\n"
end
