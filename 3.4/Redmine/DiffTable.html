def add_line(line)
  unless @parsing
    if line =~ /^(---|\+\+\+) (.*)$/
      self.file_name = $2
    elsif line =~ /^@@ (\+|\-)(\d+)(,\d+)? (\+|\-)(\d+)(,\d+)? @@/
      @line_num_l = $2.to_i
      @line_num_r = $5.to_i
      @parsing = true
    end
  else
    if line =~ %r{^[^\+\-\s@\\]}
      @parsing = false
      return false
    elsif line =~ /^@@ (\+|\-)(\d+)(,\d+)? (\+|\-)(\d+)(,\d+)? @@/
      @line_num_l = $2.to_i
      @line_num_r = $5.to_i
    else
      parse_line(line, @type)
    end
  end
  return true
end

def each_line
  prev_line_left, prev_line_right = nil, nil
  each do |line|
    spacing = prev_line_left && prev_line_right && (line.nb_line_left != prev_line_left+1) && (line.nb_line_right != prev_line_right+1)
    yield spacing, line
    prev_line_left = line.nb_line_left.to_i if line.nb_line_left.to_i > 0
    prev_line_right = line.nb_line_right.to_i if line.nb_line_right.to_i > 0
  end
end
