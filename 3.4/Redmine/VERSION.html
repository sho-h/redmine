def self.revision
  if File.directory?(File.join(Rails.root, '.svn'))
    begin
      path = Redmine::Scm::Adapters::AbstractAdapter.shell_quote(Rails.root.to_s)
      if `svn info --xml #{path}` =~ /revision="(\d+)"/
        return $1.to_i
      end
    rescue
      # Could not find the current revision
    end
  end
  nil
end
