EXTENSIONS = 
MIME_TYPES.inject({}) do |map, (type, exts)|
  exts.split(',').each {|ext| map[ext.strip] = type}
  map
end

def self.of(name)
  ext = File.extname(name.to_s)[1..-1]
  if ext
    ext.downcase!
    EXTENSIONS[ext] ||
      ((mi = MiniMime.lookup_by_extension(ext)) && mi.content_type)
  end
end
