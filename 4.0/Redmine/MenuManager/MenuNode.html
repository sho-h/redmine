def add_at(child, position)
  raise "Child already added" if find {|node| node.name == child.name}

  @children = @children.insert(position, child)
  child.parent = self
  child
end

def remove!(child)
  @children.delete(child)
  @last_items_count -= +1 if child && child.last
  child.parent = nil
  child
end
